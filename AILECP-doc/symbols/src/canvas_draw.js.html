  9 /**
 10  * Draws the map onto the canvas
 11  * @author Rahix
 12  * @param {Object} xOff Let this one be 0, you won't need this
 13  * @param {Object} yOff Let this one be 0, you won't need this
 14  */
 15 function _cpdDrawMap(xOff, yOff) 
 16 {
 17 	var tiledim;
 18 
 19 
 20 	//Get tile dimension
 21 	tiledim = cpGetTileDimension();
 22 	//Now Start to Draw
 23 	var i;
 24 	var t;
 25 	for ( i = 0; i < _cpMapData.General.Size.X; i++) 
 26 	{
 27 		for ( t = 0; t < _cpMapData.General.Size.Y; t++) 
 28 		{
 29 			if (_cpMap[t][i].Tile < 15) 
 30 			{
 31 				_cpCCTX.drawImage(document.getElementById("img_tile"), i * tiledim, t * tiledim, tiledim, tiledim);
 32 			}
 33 			else if(_cpMap[t][i].Tile == 17) //Start
 34 			{
 35 				_cpdDrawWayArrow(tiledim,i,t);
 36 				_cpCCTX.drawImage(document.getElementById(_cpcImgStartTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 37 			}
 38 			else if(_cpMap[t][i].Tile == 16)//End
 39 			{
 40 				_cpdDrawWayArrow(tiledim,i,t);
 41 				_cpCCTX.drawImage(document.getElementById(_cpcImgEndTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 42 			}
 43 			else if(_cpMap[t][i].Tile == 15)//Way
 44 			{
 45 				_cpdDrawWayArrow(tiledim,i,t);
 46 			}
 47 		}
 48 	}
 49 }

 52 /**
 53  * Draw a way arrow.
 54  * @param {Object} tiledim Tiledimension of tiles
 55  * @param {Object} i Y coordinate of way tile
 56  * @param {Object} t X coordinate of way tile
 57  */
 58 function _cpdDrawWayArrow(tiledim, i, t) 
 59 {
 60 	if (_cpMap[t][i].Dir == "N") 
 61 	{
 62 		_cpCCTX.drawImage(document.getElementById(_cpcImgUpTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 63 	}
 64 	if (_cpMap[t][i].Dir == "E") 
 65 	{
 66 		_cpCCTX.drawImage(document.getElementById(_cpcImgRightTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 67 	}
 68 	if (_cpMap[t][i].Dir == "S") 
 69 	{
 70 		_cpCCTX.drawImage(document.getElementById(_cpcImgDownTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 71 	}
 72 	if (_cpMap[t][i].Dir == "W") 
 73 	{
 74 		_cpCCTX.drawImage(document.getElementById(_cpcImgLeftTileID), i * tiledim, t * tiledim, tiledim, tiledim);
 75 	}
 76 }

 79 /**
 80  * Erases the screen
 81  * @author Rahix
 82  */
 83 function _cpdErase()
 84 {
 85 	_cpCCTX.fillRect(0,0,_cpCanvasObj.width,_cpCanvasObj.height);
 86 }
 87 
 88 /**
 89  * Draws everything, calls _cpdDrawMap
 90  * @author Rahix
 91  */
 92 function _cpdDraw() 
 93 {
 94 	_cpdErase();
 95 	_cpdDrawMap(0, 0);
 96 }
